Phase I:
GET preconditions:
	1. File must exist
	2. Can't request a directory
POST preconditions:
	1. Can't create same file in location twice

enum	ERROR, GET

httpRequest
	receive request
	call parse(first) -> TYPE, pathname or ERROR
	if ERROR 
		return 404
	if TYPE == GET
		call handleGET(pathname) -> status
	if TYPE == POST
		call parse(GET) -> content, length
		call handlePOST(pathname, content, length) -> status
	return status

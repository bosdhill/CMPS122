Phase I:
GET preconditions:
	1. File must exist
	2. Can't request a directory
POST preconditions:
	1. Can't create same file in location twice

httpRequest
	receive request
	parse first line of request to see if its POST or GET
	store pathname of request
	store content of request
	if GET
		call handleGET(pathname) -> status
	if POST
		call handlePOST(pathname, content) -> status
	return status
